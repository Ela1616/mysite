Калькулятор жобасы
Калькулятор жобасы
# Қарапайым калькулятор
def calc():
    a = float(input("1-сан: "))
    b = float(input("2-сан: "))
    op = input("Операция (+, -, *, /): ")
    if op == "+":
        return a + b
    elif op == "-":
        return a - b
    elif op == "*":
        return a * b
    elif op == "/":
        return a / b
    else:
        return "Қате операция!"

print("Нәтиже:", calc())

Telegram бот жобасы
Telegram бот жобасы
# Python Telegram Bot мысалы
from telegram import Update
from telegram.ext import Updater, CommandHandler, CallbackContext

def start(update: Update, context: CallbackContext):
    update.message.reply_text("Сәлем! Мен ботпын.")

updater = Updater("TELEGRAM_BOT_TOKEN")
updater.dispatcher.add_handler(CommandHandler("start", start))
updater.start_polling()
updater.idle()
